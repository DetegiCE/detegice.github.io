* 라벨을 이용한 return

```kotlin
package chap04.section3.noinline

fun main() {
    retFunc()
}

inline fun inlineLambda(a: Int, b: Int, out: (Int, Int) -> Unit) {
    out(a, b)
}

fun retFunc() {
    println("start of retFunc")
    inlineLambda(13, 3) lit@{ a, b ->
        val result = a + b
        if(result > 10) return@lit
        println("result: $result")
    }
    println("end of retFunc")
}
```

* 라벨을 이용한 break, continue

```kotlin
package chap04.section3

fun main() {
    println("Break")
    first@ for(i in 1..5) {
        second@ for(j in 1..5) {
            if(j == 3) break@first
            println("i:$i, j:$j")
        }
        println("after for j")
    }
    println("after for i")

    println("Continue")
    first@ for(i in 1..5) {
        second@ for(j in 1..5) {
            if(j == 3) continue@first
            println("i$i, j:$j")
        }
        println("after for j")
    }
    println("after for i")
}
```

```

i:1, j:1
i:1, j:2
after for i
Continue
i1, j:1
i1, j:2
i2, j:1
i2, j:2
i3, j:1
i3, j:2
i4, j:1
i4, j:2
i5, j:1
i5, j:2
after for i
```

* try-catch

```kotlin
package chap04.section3

fun main() {
    var a = 6
    var b = 0
    val c: Int

    try {
        c = a / b
    } catch (e: Exception) {
        println("Exception handled ${e.message}")
        e.printStackTrace()
    } finally {
        println("finally executed")
    }
}
```

```

finally executed

	at chap04.section3.TryCatchKt.main(TryCatch.kt:9)

```

* Throw Exception

```kotlin
package chap04.section3

fun main() {
    var amount = 600

    try {
        amount -= 100
        checkAmount(amount)
    } catch(e: Exception) {
        println(e.message)
    }
    print("amount: $amount")
}

fun checkAmount(amount: Int) {
    if(amount < 1000) {
        throw Exception("잔고가 $amount 로 1000 이하 입니다.")
    }
}
```

```
잔고가 500 로 1000 이하 입니다.
amount: 500
```

* Custom Exception 

```kotlin
package chap04.section3

class InvalidNameException(message: String): Exception(message)

fun main() {
    var name = "Kildong123"
    try {
        validateName(name)
    } catch(e: Exception) {
        println(e.message)
    } 
}

fun validateName(name: String) {
    if(name.matches(Regex(".*\\d+.*"))) {
        throw InvalidNameException("Your name : $name : contains numerals")
    }
}
```

```
Your name : Kildong123 : contains numerals
```
